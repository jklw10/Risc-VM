// ==========================================
// The Prepositional Pipeline Language Test
// ==========================================

// 1. Basic Variables & Arithmetic
base = 10;
multiplier = 2;
offset = base * multiplier; // 20

// 2. Memory Allocation (`alloc[:size][type]`)
data = alloc[:10][int];

// 3. Pipeline Functions (Input -> Context -> Output)
// "sqr is with x of int, to where { ... } to result"
sqr = (x[int]) : {
    result = x * x;
} : result;

// 4. Array Access
data[0] = sqr(5); // 25
data[1] = offset; // 20

// 5. Pipeline Loops (Mapping across memory)
// "with 2 to 5 of i, to where { ... }"
// Note: Loop runs from start up to (but not including) end.
(2:5)[i] : {
    data[i] = i * 10; // data[2]=20, data[3]=30, data[4]=40
};

// ==========================================
// 6. Data-Oriented Design System Example
// ==========================================
pos = alloc[:5][int];
vel = alloc[:5][int];

// Initialize DOD arrays
(0:5)[j] : {
    pos[j] = j * 5;  // pos[4] = 20
    vel[j] = 2;      // vel[4] = 2
};

// Define the physics system pipeline
physics_system = (dt[int]) : {
    
    // Bulk process over memory
    (0:5)[k] : {
        pos[k] = pos[k] + (vel[k] * dt);
    };

} : (); // Empty return type

// Execute system with delta time of 3
physics_system(3);
// Expected pos[4] -> originally 20 + (2 * 3) = 26


// ==========================================
// 7. Memory-Mapped I/O Output
// ==========================================
// The Python emulator listens at memory address 65000.
// Since arrays are 4-byte integers, index 16250 = address 65000.
screen = 0;
screen[16250] = pos[4]; // Should output 26 to the console!
screen[65000] = 404; // Should output 26 to the console!
